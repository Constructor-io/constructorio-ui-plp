# `UseSearchResults`
A React Hook to retrieve search results using Constructor.

```jsx
import { CioPlp, SearchResults } from '@constructor-io/constructorio-ui-plp'

function SearchResults({ query }) {
  const { data } = useSearchResults(query);

  return <ul>{data.response?.results.map((result) => <li>{result.itemName}</li>)}</ul>
}

function myApp() {
  ...
  return (
    <>
    <CioPlp apiKey={MY_API_KEY}>
      <SearchResults query={query} />
    </CioPlp>
    </>
  )
}
```
