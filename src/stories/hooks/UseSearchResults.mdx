## `useSearchResults`
A React Hook to retrieve search results using Constructor.

>Note: useSearchResults can be only called in components that are children of `<CioPlp />`.

```jsx
import { CioPlp } from '@constructor-io/constructorio-ui-plp'

function MySearchResults() {
  const { data } = useSearchResults({ query: 'red', searchParams: { page: 1 } });

  return <ul>{data.response?.results.map((result) => <li>{result.itemName}</li>)}</ul>
}

function myApp() {
  ...
  return (
    <CioPlp apiKey={MY_API_KEY}>
      <MySearchResults />
    </CioPlp>
  )
}
```
